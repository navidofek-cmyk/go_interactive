🏃 Závodění — goroutiny

📄 09_goroutines.go ⭐⭐⭐⭐ Těžší ⏱ ~30 minut

🤔 Co se naučíš?

Goroutina je jako kamarád, kterého pošleš dělat práci. Ty děláš svou práci a kamarád dělá svou najednou! Go může mít tisíce goroutin zároveň — to je jeho superschopnost. Klíčové slovo go před funkcí ji spustí jako goroutinu.
💡 Goroutiny = tým kamarádů. Vy stavíte LEGO hrad — jeden staví věž, druhý bránu, třetí příkop. Všichni pracují najednou, hotovo je rychleji!
package main

import (
    "fmt"
    "sync"
    "time"
)

// Pracovník dělá svou práci
func pracovnik(jmeno string, ukol string, wg *sync.WaitGroup) {
    defer wg.Done()  // oznámí "jsem hotov!" při ukončení

    fmt.Printf("🏃 %s začíná: %s\n", jmeno, ukol)
    time.Sleep(time.Second)  // simulujeme práci (1 sekunda)
    fmt.Printf("✅ %s dokončil: %s\n", jmeno, ukol)
}

func main() {
    fmt.Println("=== Stavíme LEGO hrad ===\n")

    // Bez goroutin — jeden po druhém (pomalu!)
    fmt.Println("--- Způsob 1: Sám (pomalu) ---")
    start := time.Now()

    fmt.Println("🏗️  Stavím věž...")
    time.Sleep(time.Second)
    fmt.Println("✅ Věž hotová!")

    fmt.Println("🏗️  Stavím bránu...")
    time.Sleep(time.Second)
    fmt.Println("✅ Brána hotová!")

    fmt.Printf("Čas: %v\n\n", time.Since(start))

    // S goroutinami — tým pracuje najednou (rychle!)
    fmt.Println("--- Způsob 2: Tým goroutin (rychle!) ---")
    start2 := time.Now()

    var wg sync.WaitGroup

    ukoly := []struct{ jmeno, ukol string }{
        {"Tomáš", "staví věž"},
        {"Marie", "staví bránu"},
        {"Karel", "kopí příkop"},
        {"Eva",   "maluje vlajku"},
    }

    for _, u := range ukoly {
        wg.Add(1)
        go pracovnik(u.jmeno, u.ukol, &wg)
    }

    wg.Wait()  // čekáme dokud všichni neskončí
    fmt.Printf("Čas: %v — mnohem rychleji! 🚀\n", time.Since(start2))
    fmt.Println("\nHrad je postaven! 🏰🎉")
}
📺 Co program vypíše:
=== Stavíme LEGO hrad === --- Způsob 1: Sám (pomalu) --- 🏗️ Stavím věž... ✅ Věž hotová! 🏗️ Stavím bránu... ✅ Brána hotová! Čas: 2.001s --- Způsob 2: Tým goroutin (rychle!) --- 🏃 Tomáš začíná: staví věž 🏃 Marie začíná: staví bránu 🏃 Karel začíná: kopí příkop 🏃 Eva začíná: maluje vlajku ✅ Tomáš dokončil: staví věž ✅ Marie dokončil: staví bránu ✅ Karel dokončil: kopí příkop ✅ Eva dokončil: maluje vlajku Čas: 1.001s — mnohem rychleji! 🚀 Hrad je postaven! 🏰🎉
