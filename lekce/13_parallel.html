Lekce 13: 🏭 Továrna — paralelní zpracování

📄 13_parallel.go ⭐⭐⭐⭐⭐ Výzva! ⏱ ~40 minut

🤔 Co se naučíš?

Go umí dělat víc věcí opravdu najednou — jako továrna kde každý dělník dělá svou práci zároveň. Dnes se naučíš dva slavné vzory:

  • Worker Pool — skupina dělníků (goroutin) zpracovává frontu úkolů
  • Pipeline — výrobní linka kde každý stupeň předává práci dalšímu
💡 Worker Pool = Pekárna s 3 kuchaři. Do fronty přijdou objednávky dortů. Každý kuchař vezme objednávku, upeče dort a vezme další — všichni pekou najednou!

💡 Pipeline = Výrobní linka aut: 1. svař karoserii → 2. nalak → 3. přidej motor → 4. hotovo. Každý krok běží paralelně na jiném autě zároveň.
🍰 Část 1: Worker Pool — Pekárna s kuchaři
Jak funguje Worker Pool:
Objednávky (jobs channel): 🍰→🎂→🧁→🍩→🥧→🍮 ↓ ↓ ↓ 👨‍🍳 Kuchař 1 [peče]──────→ hotové dorty 👨‍🍳 Kuchař 2 [peče]──────→ (results channel) 👨‍🍳 Kuchař 3 [peče]──────→
// Soubor: 13_worker_pool.go
// Spusť: go run 13_worker_pool.go

package main

import (
	"fmt"
	"sync"
	"time"
)

// Kuchař (worker) — vezme objednávku z fronty a upeče dort
func kuchar(id int, objednavky <-chan string, hotove chan<- string, wg *sync.WaitGroup) {
	defer wg.Done()
	for dort := range objednavky { // čte dokud fronta není prázdná
		fmt.Printf("  👨‍🍳 Kuchař %d peče: %s\n", id, dort)
		time.Sleep(300 * time.Millisecond) // pečení trvá čas
		hotove <- fmt.Sprintf("✅ %s (od kuchaře %d)", dort, id)
	}
}

func main() {
	dorty := []string{
		"🍰 Čokoládový dort",
		"🎂 Narozeninový dort",
		"🧁 Muffin s borůvkami",
		"🍩 Donut s glazurou",
		"🥧 Jablečný koláč",
		"🍮 Crème brûlée",
	}

	// Kanál pro objednávky a hotové výrobky
	objednavky := make(chan string, len(dorty))
	hotove := make(chan string, len(dorty))

	// Naplníme frontu objednávkami
	for _, d := range dorty {
		objednavky <- d
	}
	close(objednavky) // žádné další objednávky

	// Spustíme 3 kuchaře najednou (worker pool)
	var wg sync.WaitGroup
	pocetKucharu := 3

	start := time.Now()
	fmt.Printf("🏭 Pekárna otevřena! %d kuchařů, %d objednávek\n\n", pocetKucharu, len(dorty))

	for i := 1; i <= pocetKucharu; i++ {
		wg.Add(1)
		go kuchar(i, objednavky, hotove, &wg)
	}

	// Goroutina zavře hotove channel až všichni skončí
	go func() {
		wg.Wait()
		close(hotove)
	}()

	// Sbíráme hotové dorty
	fmt.Println("📦 Hotové objednávky:")
	for dort := range hotove {
		fmt.Println(" ", dort)
	}

	fmt.Printf("\n⏱  Celkový čas: %v\n", time.Since(start))
	fmt.Printf("💡 Bez paralelizace by to trvalo: %v\n",
		time.Duration(len(dorty))*300*time.Millisecond)
}

🏗️ Část 2: Pipeline — Výrobní linka
Jak funguje Pipeline:
Suroviny → [Umyj] → [Nakrájej] → [Uvař] → [Zabal] → Hotový výrobek ↑goroutina↑ ↑goroutina↑ ↑goroutina↑ ↑goroutina↑ Každý stupeň běží paralelně — zpracovává jiný kus zároveň!
// Soubor: 13_pipeline.go
// Spusť: go run 13_pipeline.go

package main

import "fmt"

// Stupeň 1: Přijme seznam surovin a pošle je dál
func prijem(suroviny []string) <-chan string {
	out := make(chan string)
	go func() {
		for _, s := range suroviny {
			out <- s
		}
		close(out)
	}()
	return out
}

// Stupeň 2: Umyje každou surovinu
func umyj(in <-chan string) <-chan string {
	out := make(chan string)
	go func() {
		for s := range in {
			out <- "🫧 " + s + " (umyto)"
		}
		close(out)
	}()
	return out
}

// Stupeň 3: Nakrájí
func nakraj(in <-chan string) <-chan string {
	out := make(chan string)
	go func() {
		for s := range in {
			out <- "🔪 " + s + " (nakrájeno)"
		}
		close(out)
	}()
	return out
}

// Stupeň 4: Zabalí do krabičky
func zabal(in <-chan string) <-chan string {
	out := make(chan string)
	go func() {
		for s := range in {
			out <- "📦 " + s + " (zabaleno)"
		}
		close(out)
	}()
	return out
}

func main() {
	suroviny := []string{"🍎 Jablko", "🍌 Banán", "🍓 Jahoda", "🥝 Kiwi"}

	fmt.Println("🏗️  Spouštíme výrobní linku!")
	fmt.Println()

	// Sestavíme pipeline — každá funkce vrátí channel
	// a přijme channel od předchozího stupně
	krok1 := prijem(suroviny)
	krok2 := umyj(krok1)
	krok3 := nakraj(krok2)
	krok4 := zabal(krok3)

	// Čteme výsledky z posledního stupně
	fmt.Println("Hotové výrobky:")
	for vyrobek := range krok4 {
		fmt.Println(" ", vyrobek)
	}

	fmt.Println()
	fmt.Println("🎉 Výrobní linka hotová!")
}

🏗️ Spouštíme výrobní linku! Hotové výrobky: 📦 🫧 🔪 🍎 Jablko (umyto) (nakrájeno) (zabaleno) 📦 🫧 🔪 🍌 Banán (umyto) (nakrájeno) (zabaleno) 📦 🫧 🔪 🍓 Jahoda (umyto) (nakrájeno) (zabaleno) 📦 🫧 🔪 🥝 Kiwi (umyto) (nakrájeno) (zabaleno) 🎉 Výrobní linka hotová!
