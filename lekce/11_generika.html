🧬 Kouzelné funkce — generika

📄 11_generika.go ⭐⭐⭐⭐ Těžší ⏱ ~30 minut

🤔 Co se naučíš?

Generická funkce je kouzelná — funguje s čísly, textem, i čímkoliv jiným! Normálně musíš psát zvláštní funkci pro každý typ. S generikami stačí jedna! Typ T je jako zástupné místo — Go ho vyplní správným typem automaticky.
💡 Generika = kouzelný universální dálkový ovladač. Jeden ovladač ovládá TV, rádio, DVD přehrávač — všechno! Bez generiky bys potřeboval zvláštní ovladač pro každé zařízení.
package main

import "fmt"

// Normální funkce — jen pro int
func nejvetsiInt(a int, b int) int {
    if a > b { return a }
    return b
}

// Generická funkce — funguje pro COKOLIV co lze porovnat!
// [T int | float64 | string] znamená "T může být jeden z těchto typů"
func nejvetsi[T int | float64 | string](a T, b T) T {
    if a > b { return a }
    return b
}

// Generický zásobník (stack) — jako věž z kostek
type Vez[T any] struct {
    kostky []T
}

func (v *Vez[T]) Poloz(kostka T) {
    v.kostky = append(v.kostky, kostka)
    fmt.Printf("  📦 Položil: %v\n", kostka)
}

func (v *Vez[T]) Vezmi() T {
    n := len(v.kostky)
    kostka := v.kostky[n-1]
    v.kostky = v.kostky[:n-1]
    fmt.Printf("  🤏 Vzal:    %v\n", kostka)
    return kostka
}

func (v *Vez[T]) Vyska() int { return len(v.kostky) }

func main() {
    fmt.Println("=== Kouzelná funkce nejvetsi ===")
    fmt.Println("Větší číslo (3, 7):", nejvetsi(3, 7))
    fmt.Println("Větší číslo (3.14, 2.71):", nejvetsi(3.14, 2.71))
    fmt.Println("Větší slovo:", nejvetsi("jablko", "pomeranč"))
    fmt.Println()

    fmt.Println("=== Věž z číselných kostek ===")
    ciselnaVez := Vez[int]{}
    ciselnaVez.Poloz(1)
    ciselnaVez.Poloz(2)
    ciselnaVez.Poloz(3)
    fmt.Printf("Výška věže: %d\n\n", ciselnaVez.Vyska())

    ciselnaVez.Vezmi()
    ciselnaVez.Vezmi()
    fmt.Printf("Výška věže po odebrání: %d\n\n", ciselnaVez.Vyska())

    fmt.Println("=== Věž ze slov ===")
    slovniVez := Vez[string]{}
    slovniVez.Poloz("🍎 jablko")
    slovniVez.Poloz("🍌 banán")
    slovniVez.Poloz("🍇 hrozny")
    fmt.Println()
    slovniVez.Vezmi()

    fmt.Println("\nSkvělé! Generika jsou opravdu kouzelné! 🧬🎉")
}
📺 Co program vypíše:
=== Kouzelná funkce nejvetsi === Větší číslo (3, 7): 7 Větší číslo (3.14, 2.71): 3.14 Větší slovo: pomeranč === Věž z číselných kostek === 📦 Položil: 1 📦 Položil: 2 📦 Položil: 3 Výška věže: 3 🤏 Vzal: 3 🤏 Vzal: 2 Výška věže po odebrání: 1 === Věž ze slov === 📦 Položil: 🍎 jablko 📦 Položil: 🍌 banán 📦 Položil: 🍇 hrozny 🤏 Vzal: 🍇 hrozny Skvělé! Generika jsou opravdu kouzelné! 🧬🎉

✏️ Zkus to sám!

  1. Zavolej nejvetsi("ahoj", "čau") — které slovo je větší?
    🔍 Zobrazit nápovědu
    fmt.Println(nejvetsi("ahoj", "čau"))
// Vypíše "čau" — písmeno č je v abecedě za a, takže "čau" > "ahoj"
  2. Vytvoř věž z bool hodnot a vlož true, false, true.
    🔍 Zobrazit nápovědu
    boolVez := Vez[bool]{}
boolVez.Poloz(true)
boolVez.Poloz(false)
boolVez.Poloz(true)
  3. Přidej do Vez metodu JePrazdna() bool která vrátí true pokud je věž prázdná.
    🔍 Zobrazit nápovědu
    func (v *Vez[T]) JePrazdna() bool {
    return len(v.kostky) == 0
}
