Lekce 06: 🧩 Postavičky — struktury

📄 06_struktury.go ⭐⭐⭐ Střední ⏱ ~30 minut

🤔 Co se naučíš?

Struktura (struct) je jako hráčská karta ve videohře. Na kartě jsou různé informace dohromady: jméno, životy, level, zbraň. Všechno popisuje jednu věc — jednoho hráče. Ke struktuře pak přidáme metody — to jsou akce, které hráč umí dělat.

💡 Struct = Pokémon karta. Každá karta má: jméno, HP, typ, útok. Struct ti umožní vytvořit libovolnou vlastní "kartu" pro cokoliv — hráče, zvíře, auto, planetu!
// Soubor: 06_struktury.go
// Spusť: go run 06_struktury.go

package main

import "fmt"

// Hráčská karta — struct
type Hrac struct {
    Jmeno  string
    Zivoty int
    Level  int
    Zbran  string
}

// Metoda — Hrac umí vypsat svou kartu
func (h Hrac) VypisKartu() {
    fmt.Println("┌─────────────────────────┐")
    fmt.Printf( "│ 🎮 %-22s│\n", h.Jmeno)
    fmt.Printf( "│ ❤️  Životy: %-13d│\n", h.Zivoty)
    fmt.Printf( "│ ⭐ Level:   %-13d│\n", h.Level)
    fmt.Printf( "│ ⚔️  Zbraň:  %-13s│\n", h.Zbran)
    fmt.Println("└─────────────────────────┘")
}

// Metoda která mění hráče — potřebuje * (ukazatel)
func (h *Hrac) SeberePoklad(body int) {
    fmt.Printf("🎁 %s sebral poklad! +%d level\n", h.Jmeno, body)
    h.Level += body
}

func (h *Hrac) UtrpiPoskozeni(skoda int) {
    h.Zivoty -= skoda
    if h.Zivoty < 0 {
        h.Zivoty = 0
    }
    fmt.Printf("💥 %s dostal %d poškození! Životy: %d\n", h.Jmeno, skoda, h.Zivoty)
}

func main() {
    // Vytvoříme hráče
    hrac1 := Hrac{
        Jmeno:  "Tomáš",
        Zivoty: 100,
        Level:  1,
        Zbran:  "meč",
    }

    hrac2 := Hrac{
        Jmeno:  "Marie",
        Zivoty: 80,
        Level:  3,
        Zbran:  "luk",
    }

    fmt.Println("=== Hráčské karty ===")
    hrac1.VypisKartu()
    fmt.Println()
    hrac2.VypisKartu()
    fmt.Println()

    // Simulujeme hru!
    fmt.Println("=== Průběh hry ===")
    hrac1.SeberePoklad(5)
    hrac1.UtrpiPoskozeni(30)
    hrac2.SeberePoklad(2)
    hrac2.UtrpiPoskozeni(10)

    fmt.Println("\n=== Stav po boji ===")
    hrac1.VypisKartu()
    fmt.Println()
    hrac2.VypisKartu()

    fmt.Println("\nSkvělé! Umíš vytvářet herní postavičky! 🎉")
}
