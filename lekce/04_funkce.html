Lekce 04: 🔧 Funkce — vlastní příkazy

📄 04_funkce.go ⭐⭐ Snadné ⏱ ~25 minut

🤔 Co se naučíš?

Funkce je jako recept. Napíšeš recept jednou a pak ho můžeš použít kolikrát chceš! Třeba funkce pozdrav — zavoláš ji a funkce pozdraví. Nemusíš psát postup znovu a znovu. Funkce mohou dostávat vstupy (co jim dáš) a vracet výsledek (co ti vrátí zpět).

💡 Funkce = recept na jídlo. Napíšeš ho jednou, kuchař ho může použít 100×. Nebo tlačítko na dálkovém ovladači — stiskneš a TV se zapne. Pokaždé stejný výsledek!
// Soubor: 04_funkce.go
// Spusť: go run 04_funkce.go

package main

import "fmt"

// Funkce bez návratové hodnoty
func pozdrav(jmeno string) {
    fmt.Printf("Ahoj, %s! 👋\n", jmeno)
}

// Funkce s návratovou hodnotou
func secti(a int, b int) int {
    return a + b
}

// Funkce vrací dvě hodnoty najednou!
func minMax(a int, b int) (int, int) {
    if a < b {
        return a, b
    }
    return b, a
}

// Funkce kreslí zdravotní bar jako ve hře
func zdraviBar(aktualni int, max int) {
    fmt.Print("❤️  [")
    for i := 0; i < max; i++ {
        if i < aktualni {
            fmt.Print("█")
        } else {
            fmt.Print("░")
        }
    }
    fmt.Printf("] %d/%d\n", aktualni, max)
}

func main() {
    // Voláme naše funkce!
    pozdrav("Tomáš")
    pozdrav("Marie")
    pozdrav("Robot Go")
    fmt.Println()

    vysledek := secti(5, 3)
    fmt.Println("5 + 3 =", vysledek)
    fmt.Println("10 + 25 =", secti(10, 25))
    fmt.Println()

    mensi, vetsi := minMax(42, 17)
    fmt.Println("Menší číslo:", mensi)
    fmt.Println("Větší číslo:", vetsi)
    fmt.Println()

    // Zdravotní bar jako ve hře!
    fmt.Println("=== Zdraví hráčů ===")
    zdraviBar(10, 10) // plné zdraví
    zdraviBar(7, 10)  // 70% zdraví
    zdraviBar(3, 10)  // kritické!
    zdraviBar(0, 10)  // mrtvý

    fmt.Println("\nSkvělé! Umíš psát vlastní funkce! 🎉")
}
